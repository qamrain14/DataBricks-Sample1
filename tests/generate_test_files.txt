files["dlt_mock.py"] = r'''"""
DLT Mock Framework for offline testing of Databricks DLT pipelines.
Simulates dlt.table, dlt.read, dlt.expect, dlt.expect_or_drop, dlt.expect_or_fail.
"""

import functools

# Internal registries
_tables = {}
_table_metadata = {}
_expectation_results = {}


def reset():
    """Clear all registered tables and expectation results."""
    _tables.clear()
    _table_metadata.clear()
    _expectation_results.clear()


def register(name, df):
    """Register a DataFrame as a named DLT table."""
    _tables[name] = df


def read(name):
    """Read a registered DLT table by name."""
    if name not in _tables:
        available = list(_tables.keys())
        raise KeyError(f"Table '{name}' not registered in dlt_mock. Available: {available}")
    return _tables[name]


def get_table(name):
    """Get a registered table or None."""
    return _tables.get(name)


def get_all_tables():
    """Return a copy of all registered tables."""
    return dict(_tables)


def get_expectations():
    """Return all recorded expectation results."""
    return dict(_expectation_results)


def table(name=None, comment=None, table_properties=None, path=None):
    """Decorator simulating @dlt.table - registers the returned DataFrame."""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            df = func(*args, **kwargs)
            tbl_name = name or func.__name__
            _tables[tbl_name] = df
            _table_metadata[tbl_name] = {
                "comment": comment,
                "table_properties": table_properties,
                "path": path,
            }
            return df
        return wrapper
    return decorator


def expect(rule_name, condition):
    """Decorator simulating @dlt.expect - warns but keeps all rows."""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            df = func(*args, **kwargs)
            violations = df.where(f"NOT ({condition})").count()
            tbl_name = _find_table_name(df, rule_name)
            _expectation_results.setdefault(tbl_name, []).append(
                (rule_name, condition, violations, "warn")
            )
            return df
        return wrapper
    return decorator


def expect_or_drop(rule_name, condition):
    """Decorator simulating @dlt.expect_or_drop - filters violating rows."""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            df = func(*args, **kwargs)
            before = df.count()
            filtered = df.where(condition)
            after = filtered.count()
            dropped = before - after
            tbl_name = _find_table_name(df, rule_name)
            _expectation_results.setdefault(tbl_name, []).append(
                (rule_name, condition, dropped, "drop")
            )
            # Re-register the filtered DataFrame
            for n, d in list(_tables.items()):
                if d is df:
                    _tables[n] = filtered
                    break
            return filtered
        return wrapper
    return decorator


def expect_or_fail(rule_name, condition):
    """Decorator simulating @dlt.expect_or_fail - raises on violations."""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            df = func(*args, **kwargs)
            violations = df.where(f"NOT ({condition})").count()
            if violations > 0:
                raise ValueError(
                    f"DLT expect_or_fail FAILED: rule='{rule_name}', "
                    f"condition='{condition}', violations={violations}"
                )
            tbl_name = _find_table_name(df, rule_name)
            _expectation_results.setdefault(tbl_name, []).append(
                (rule_name, condition, 0, "fail")
            )
            return df
        return wrapper
    return decorator


def _find_table_name(df, default):
    """Find the registered table name for a DataFrame."""
    for n, d in _tables.items():
        if d is df:
            return n
    return default
'''
