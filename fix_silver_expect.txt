"""Fix silver expect predicates that reference renamed-away source columns."""
from pathlib import Path
p = Path(__file__).resolve().parent / "databricks" / "silver" / "02_silver_pipeline.py"
text = p.read_text(encoding="utf-8")
old = '@dlt.expect("valid_amount", "gross_amount > 0")'
new = '@dlt.expect("valid_amount", "invoice_amount > 0")'
assert old in text
text = text.replace(old, new)
p.write_text(text, encoding="utf-8")
print(f"Patched silver expect; size={p.stat().st_size}")
